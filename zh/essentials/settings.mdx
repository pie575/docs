---
title: '全局设置'
description: 'Mintlify 通过 docs.json 配置文件让你完全掌控文档的外观与体验'

---

每个 Mintlify 站点都需要一个包含核心配置的 `docs.json` 文件。请在下方了解更多[属性](#properties)。

<div id="properties">
  ## 属性
</div>

<ResponseField name="name" type="string" required>
  项目名称。用于全局标题。

  示例：`mintlify`
</ResponseField>

<ResponseField name="navigation" type="Navigation[]" required>
  包含各分组内所有页面的分组数组

  <Expandable title="Navigation">
    <ResponseField name="group" type="string">
      分组名称。

      示例：`Settings`
    </ResponseField>

    <ResponseField name="pages" type="string[]">
      作为页面使用的 Markdown 文件相对路径。

      示例：`["customization", "page"]`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="logo" type="string or object">
  指向 logo 图片的路径，或包含「light」与「dark」模式 logo 路径的对象

  <Expandable title="Logo">
    <ResponseField name="light" type="string">
      浅色模式下的 logo 路径
    </ResponseField>

    <ResponseField name="dark" type="string">
      深色模式下的 logo 路径
    </ResponseField>

    <ResponseField name="href" type="string" default="/">
      点击 logo 后跳转的链接
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  favicon 图片路径
</ResponseField>

<ResponseField name="colors" type="Colors">
  全局主题的十六进制颜色代码

  <Expandable title="Colors">
    <ResponseField name="primary" type="string" required>
      主色。常用于高亮内容、章节标题、强调色（浅色模式）
    </ResponseField>

    <ResponseField name="light" type="string">
      深色模式下的主色。常用于高亮内容、章节标题、强调色（深色模式）
    </ResponseField>

    <ResponseField name="dark" type="string">
      重要按钮的主色
    </ResponseField>

    <ResponseField name="background" type="object">
      浅色与深色模式下的背景色

      <Expandable title="Object">
        <ResponseField name="light" type="string" required>
          浅色模式下背景的十六进制颜色代码
        </ResponseField>

        <ResponseField name="dark" type="string" required>
          深色模式下背景的十六进制颜色代码
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  顶栏中要包含的链接的 `name` 和 `url` 数组

  <Expandable title="TopbarLink">
    <ResponseField name="name" type="string">
      按钮名称。

      示例：`Contact us`
    </ResponseField>

    <ResponseField name="url" type="string">
      点击按钮后的链接地址。示例：`https://mintlify.com/docs`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  <Expandable title="Topbar Call to Action">
    <ResponseField name="type" type={'"link" or "github"'} default="link">
      link 显示一个按钮；github 会展示所提供 URL 的仓库信息，包括 GitHub 星标数。
    </ResponseField>

    <ResponseField name="url" type="string">
      若为 `link`：按钮跳转的地址。

      若为 `github`：用于加载 GitHub 信息的仓库链接。
    </ResponseField>

    <ResponseField name="name" type="string">
      按钮文本。仅当 `type` 为 `link` 时必填。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="versions" type="string[]">
  版本名称数组。仅当你希望在导航栏中通过下拉菜单展示不同文档版本时使用。
</ResponseField>

<ResponseField name="anchors" type="Anchor[]">
  锚点数组，包含 `icon`、`color` 和 `url`。

  <Expandable title="Anchor">
    <ResponseField name="icon" type="string">
      用于展示锚点的 [Font Awesome](https://fontawesome.com/search?q=heart) 图标。

      示例：`comments`
    </ResponseField>

    <ResponseField name="name" type="string">
      锚点标签名称。

      示例：`Community`
    </ResponseField>

    <ResponseField name="url" type="string">
      标记哪些页面归属该锚点的 URL 起始段。通常为你存放页面的文件夹名称。
    </ResponseField>

    <ResponseField name="color" type="string">
      锚点图标背景的十六进制颜色。若传入一个包含 `from` 和 `to` 属性的对象（各为十六进制颜色），也可设置为渐变。
    </ResponseField>

    <ResponseField name="version" type="string">
      当你希望在选择正确的文档版本之前隐藏该锚点时使用。
    </ResponseField>

    <ResponseField name="isDefaultHidden" type="boolean" default="false">
      如需在直接将他人链接到其内部文档之前隐藏该锚点，请传入 `true`。
    </ResponseField>

    <ResponseField name="iconType" default="duotone" type="string">
      可选值：&quot;brands&quot;、&quot;duotone&quot;、&quot;light&quot;、&quot;sharp-solid&quot;、&quot;solid&quot;、&quot;thin&quot;
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topAnchor" type="Object">
  覆盖最上层锚点的默认配置。

  <Expandable title="Object">
    <ResponseField name="name" default="Documentation" type="string">
      最上层锚点名称
    </ResponseField>

    <ResponseField name="icon" default="book-open" type="string">
      Font Awesome 图标。
    </ResponseField>

    <ResponseField name="iconType" default="duotone" type="string">
      可选值：&quot;brands&quot;、&quot;duotone&quot;、&quot;light&quot;、&quot;sharp-solid&quot;、&quot;solid&quot;、&quot;thin&quot;
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tabs" type="Tabs[]">
  导航选项卡数组。

  <Expandable title="Tabs">
    <ResponseField name="name" type="string">
      选项卡标签名称。
    </ResponseField>

    <ResponseField name="url" type="string">
      标记哪些页面归属该选项卡的 URL 起始段。通常这是你存放页面的文件夹名称。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="api" type="API">
  API 设置相关的配置。参见 [API Components](/zh/api-playground/demo) 以了解更多 API 页面信息。

  <Expandable title="API">
    <ResponseField name="baseUrl" type="string">
      所有 API 端点的基础 URL。若 `baseUrl` 为数组，则可提供多个基础 URL 选项，供用户切换。
    </ResponseField>

    <ResponseField name="auth" type="Auth">
      <Expandable title="Auth">
        <ResponseField name="method" type="&#x22;bearer&#x22; | &#x22;basic&#x22; | &#x22;key&#x22;">
          用于所有 API 端点的身份验证方式。
        </ResponseField>

        <ResponseField name="name" type="string">
          在 API playground 中使用的身份验证参数名称。

          如果 method 为 `basic`，格式应为 `[usernameName]:[passwordName]`
        </ResponseField>

        <ResponseField name="inputPrefix" type="string">
          作为身份验证输入字段前缀的默认值。

          例如，若 `inputPrefix` 为 `AuthKey`，则身份验证字段的默认输入结果将为 `AuthKey`。
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="playground" type="Playground">
      API playground 的配置

      <Expandable title="Playground">
        <ResponseField name="mode" default="show" type="&#x22;show&#x22; | &#x22;simple&#x22; | &#x22;hide&#x22;">
          控制 playground 的展示模式：显示、隐藏，或仅显示端点且无用户交互（`simple`）。

          详见 [playground 指南](/zh/api-playground/demo)
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="maintainOrder" type="boolean">
      启用后可确保 OpenAPI 页面中的键顺序与 OpenAPI 文件中定义的顺序一致。

      <Warning>此行为即将默认启用，届时该字段将被弃用。</Warning>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="openapi" type="string | string[]">
  指向你的 OpenAPI 文件的 URL 或相对路径的字符串，或其字符串数组。

  示例：

  <CodeGroup>
    ```json Absolute
    "openapi": "https://example.com/openapi.json"
    ```

    ```json Relative
    "openapi": "/openapi.json"
    ```

    ```json Multiple
    "openapi": ["https://example.com/openapi1.json", "/openapi2.json", "/openapi3.json"]
    ```
  </CodeGroup>
</ResponseField>

<ResponseField name="footerSocials" type="FooterSocials">
  一个社交媒体账号对象，其中 key:property 对应表示社交媒体平台与账号 URL。

  示例：

  ```json
  {
    "x": "https://x.com/mintlify",
    "website": "https://mintlify.com"
  }
  ```

  <Expandable title="FooterSocials">
    <ResponseField name="[key]" type="string">
      以下值之一：`website`、`facebook`、`x`、`discord`、`slack`、`github`、`linkedin`、`instagram`、`hacker-news`

      示例：`x`
    </ResponseField>

    <ResponseField name="property" type="string">
      该社交平台的 URL。

      示例：`https://x.com/mintlify`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="feedback" type="Feedback">
  启用反馈按钮的配置

  <Expandable title="Feedback">
    <ResponseField name="suggestEdit" type="boolean" default="false">
      启用按钮，允许用户通过 pull request 提出修改建议
    </ResponseField>

    <ResponseField name="raiseIssue" type="boolean" default="false">
      启用按钮，允许用户就文档提交问题
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="modeToggle" type="ModeToggle">
  自定义深色模式切换按钮。

  <Expandable title="ModeToggle">
    <ResponseField name="default" type={'"light" or "dark"'}>
      设置新用户默认显示浅色或深色模式。未设置时，默认为与用户操作系统相同的模式。
    </ResponseField>

    <ResponseField name="isHidden" type="boolean" default="false">
      设为 true 可隐藏深色/浅色模式切换按钮。你可以将 `isHidden` 与 `default` 结合使用，强制文档仅使用浅色或深色模式。例如：

      <CodeGroup>
        ```json Only Dark Mode
        "modeToggle": {
          "default": "dark",
          "isHidden": true
        }
        ```

        ```json Only Light Mode
        "modeToggle": {
          "default": "light",
          "isHidden": true
        }
        ```
      </CodeGroup>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  在每个页面背景中显示的背景图片。可参考 [Infisical](https://infisical.com/docs) 和 [FRPC](https://frpc.io) 的示例。
</ResponseField>
